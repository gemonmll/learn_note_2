```cpp

namespace apollo {
namespace cyber {
namespace base {

template <typename T>
class BoundedQueue {
 public:
  using value_type = T;
  using size_type = uint64_t;

 public:
  BoundedQueue() {}
  BoundedQueue& operator=(const BoundedQueue& other) = delete;
  BoundedQueue(const BoundedQueue& other) = delete;
  ~BoundedQueue();
  bool Init(uint64_t size);
  bool Init(uint64_t size, WaitStrategy* strategy);
  bool Enqueue(const T& element);
  bool Enqueue(T&& element);
  bool WaitEnqueue(const T& element);
  bool WaitEnqueue(T&& element);
  bool Dequeue(T* element);
  bool WaitDequeue(T* element);
  uint64_t Size();
  bool Empty();
  void SetWaitStrategy(WaitStrategy* WaitStrategy);
  void BreakAllWait();
  uint64_t Head() { return head_.load(); }
  uint64_t Tail() { return tail_.load(); }
  uint64_t Commit() { return commit_.load(); }

 private:
  uint64_t GetIndex(uint64_t num);

  alignas(CACHELINE_SIZE) std::atomic<uint64_t> head_ = {0};
  alignas(CACHELINE_SIZE) std::atomic<uint64_t> tail_ = {1};
  alignas(CACHELINE_SIZE) std::atomic<uint64_t> commit_ = {1};
  // alignas(CACHELINE_SIZE) std::atomic<uint64_t> size_ = {0};
  uint64_t pool_size_ = 0;
  T* pool_ = nullptr;
  std::unique_ptr<WaitStrategy> wait_strategy_ = nullptr;
  volatile bool break_all_wait_ = false;
};

template <typename T>
BoundedQueue<T>::~BoundedQueue() {
  if (wait_strategy_) {
    BreakAllWait();
  }
  if (pool_) {
    for (uint64_t i = 0; i < pool_size_; ++i) {
      pool_[i].~T();
    }
    std::free(pool_);
  }
}

template <typename T>
inline bool BoundedQueue<T>::Init(uint64_t size) {
  return Init(size, new SleepWaitStrategy());
}

template <typename T>
bool BoundedQueue<T>::Init(uint64_t size, WaitStrategy* strategy) {
  // Head and tail each occupy a space
  pool_size_ = size + 2;
  pool_ = reinterpret_cast<T*>(std::calloc(pool_size_, sizeof(T)));
  if (pool_ == nullptr) {
    return false;
  }
  for (uint64_t i = 0; i < pool_size_; ++i) {
    new (&(pool_[i])) T();
  }
  wait_strategy_.reset(strategy);
  return true;
}

template <typename T>
bool BoundedQueue<T>::Enqueue(const T& element) {
  uint64_t new_tail = 0;
  uint64_t old_commit = 0;
  uint64_t old_tail = tail_.load(std::memory_order_acquire);
  do {
    new_tail = old_tail + 1;
    if (GetIndex(new_tail) == GetIndex(head_.load(std::memory_order_acquire))) {
      return false;
    }
  } while (!tail_.compare_exchange_weak(old_tail, new_tail,
                                        std::memory_order_acq_rel,
                                        std::memory_order_relaxed));
  pool_[GetIndex(old_tail)] = element;
  do {
    old_commit = old_tail;
  } while (cyber_unlikely(!commit_.compare_exchange_weak(
      old_commit, new_tail, std::memory_order_acq_rel,
      std::memory_order_relaxed)));
  wait_strategy_->NotifyOne();
  return true;
}

template <typename T>
bool BoundedQueue<T>::Enqueue(T&& element) {
  uint64_t new_tail = 0;
  uint64_t old_commit = 0;
  uint64_t old_tail = tail_.load(std::memory_order_acquire);
  do {
    new_tail = old_tail + 1;
    if (GetIndex(new_tail) == GetIndex(head_.load(std::memory_order_acquire))) {
      return false;
    }
  } while (!tail_.compare_exchange_weak(old_tail, new_tail,
                                        std::memory_order_acq_rel,
                                        std::memory_order_relaxed));
  pool_[GetIndex(old_tail)] = std::move(element);
  do {
    old_commit = old_tail;
  } while (cyber_unlikely(!commit_.compare_exchange_weak(
      old_commit, new_tail, std::memory_order_acq_rel,
      std::memory_order_relaxed)));
  wait_strategy_->NotifyOne();
  return true;
}

template <typename T>
bool BoundedQueue<T>::Dequeue(T* element) {
  uint64_t new_head = 0;
  uint64_t old_head = head_.load(std::memory_order_acquire);
  do {
    new_head = old_head + 1;
    if (new_head == commit_.load(std::memory_order_acquire)) {
      return false;
    }
    *element = pool_[GetIndex(new_head)];
  } while (!head_.compare_exchange_weak(old_head, new_head,
                                        std::memory_order_acq_rel,
                                        std::memory_order_relaxed));
  return true;
}

template <typename T>
bool BoundedQueue<T>::WaitEnqueue(const T& element) {
  while (!break_all_wait_) {
    if (Enqueue(element)) {
      return true;
    }
    if (wait_strategy_->EmptyWait()) {
      continue;
    }
    // wait timeout
    break;
  }

  return false;
}

template <typename T>
bool BoundedQueue<T>::WaitEnqueue(T&& element) {
  while (!break_all_wait_) {
    if (Enqueue(std::move(element))) {
      return true;
    }
    if (wait_strategy_->EmptyWait()) {
      continue;
    }
    // wait timeout
    break;
  }

  return false;
}

template <typename T>
bool BoundedQueue<T>::WaitDequeue(T* element) {
  while (!break_all_wait_) {
    if (Dequeue(element)) {
      return true;
    }
    if (wait_strategy_->EmptyWait()) {
      continue;
    }
    // wait timeout
    break;
  }

  return false;
}

template <typename T>
inline uint64_t BoundedQueue<T>::Size() {
  return tail_ - head_ - 1;
}

template <typename T>
inline bool BoundedQueue<T>::Empty() {
  return Size() == 0;
}

template <typename T>
inline uint64_t BoundedQueue<T>::GetIndex(uint64_t num) {
  return num - (num / pool_size_) * pool_size_;  // faster than %
}

template <typename T>
inline void BoundedQueue<T>::SetWaitStrategy(WaitStrategy* strategy) {
  wait_strategy_.reset(strategy);
}

template <typename T>
inline void BoundedQueue<T>::BreakAllWait() {
  break_all_wait_ = true;
  wait_strategy_->BreakAllWait();
}

}  // namespace base
}  // namespace cyber
}  // namespace apollo

#endif  // CYBER_BASE_BOUNDED_QUEUE_H_

```

可以，我们就专门学这三块。你现在已经知道它本体是**环形数组**了，下面这三样其实是在解决“**多线程下怎么安全又高效地用这个环形数组**”的问题：原子变量 `head_ / tail_ / commit_`、CAS、以及 `WaitStrategy`。这些都直接体现在这份 `BoundedQueue` 代码里。

---

# 1. 先说整体目标

这个队列不是给单线程玩的，而是想让：

* 一个或多个线程往里写
* 一个或多个线程从里读
* 不加大锁，尽量快
* 队列空/满时还能等待

所以它需要解决三个核心问题：

1. **谁来决定下一个可写位置、可读位置**
2. **多个线程同时抢位置时，怎么不冲突**
3. **空了/满了时，线程怎么等待，不要一直空转**

于是就有了：

* `head_ / tail_ / commit_`：记录队列状态
* CAS：原子地“抢位置”
* `WaitStrategy`：负责等待和唤醒 

---

# 2. `head_ / tail_ / commit_` 分别是干什么的

## `head_`

表示“**已经消费到哪里了**”。

看 `Dequeue()`：

```cpp
uint64_t old_head = head_.load(...);
new_head = old_head + 1;
...
} while (!head_.compare_exchange_weak(old_head, new_head, ...));
```

说明出队成功后，会把 `head_` 往前推进。也就是：

> `head_` 是消费者的进度。

---

## `tail_`

表示“**生产者已经申请到哪里了**”。

看 `Enqueue()`：

```cpp
uint64_t old_tail = tail_.load(...);
new_tail = old_tail + 1;
...
} while (!tail_.compare_exchange_weak(old_tail, new_tail, ...));
```

这说明一个生产者线程想写数据时，会先把 `tail_` 抢到手，给自己占一个槽位。也就是：

> `tail_` 是写位置的分配进度。

---

## `commit_`

这是这份实现里最值得学的点。

它表示：

> **已经真正写完、消费者可以安全读取的位置边界**

因为“申请到写位置”和“数据真正写完”不是同一时刻。

`Enqueue()` 里顺序是：

1. 先 CAS 推进 `tail_`，抢到槽位
2. 再执行
   `pool_[GetIndex(old_tail)] = element;`
3. 最后再 CAS 推进 `commit_` 

所以：

* `tail_` 快一点，表示“位置被占了”
* `commit_` 慢一点，表示“数据真的写好了”

消费者判断能不能读，不看 `tail_`，而看 `commit_`：

```cpp
if (new_head == commit_.load(...)) {
  return false;
}
```

这就是在说：

> 只有 `commit_` 之前的数据，才算真的可读。

---

# 3. 为什么不能只有 `head_` 和 `tail_`

因为多线程时，可能发生这种事：

假设线程 A 正在入队：

1. A 先把 `tail_` 从 5 改到 6
2. 但 A 还没来得及把数据写进 `pool_[5]`

这时消费者线程 B 来了，如果它只看 `tail_`，就会以为：

> “哦，已经有新数据了，我可以读 5 号槽位了。”

但实际上 5 号槽位可能还没写完，B 就读到脏数据了。

所以要加一个 `commit_` 来分开：

* `tail_`：位置已分配
* `commit_`：内容已提交

这是这个实现的精髓。

---

# 4. 你可以把它想成“订座”和“上菜”

这个比喻挺贴切：

* `tail_`：相当于服务员先给客人占了桌子
* `pool_[...] = element`：厨房正在做菜
* `commit_`：菜做好端上桌，客人能吃了

消费者不是看“桌子有没有被占”，而是看“菜有没有端上来”。

---

# 5. CAS 是干什么的

CAS 全名可以理解成：

> **比较并交换（Compare And Swap / Compare And Exchange）**

在这份代码里用的是：

```cpp
compare_exchange_weak(...)
```

它干的事很简单：

1. 先看当前值是不是我以为的旧值
2. 如果是，就改成新值
3. 如果不是，说明别人抢先改了，我失败，重试

---

## 为什么要用 CAS

因为可能有多个生产者线程同时入队。

比如两个线程都看到：

```cpp
old_tail = 10
```

它们都想占下一个槽位 10。

如果没有原子操作，可能会发生：

* 线程 A 觉得自己拿到 10
* 线程 B 也觉得自己拿到 10

然后两个线程往同一个槽位写，数据就乱了。

CAS 就是为了保证：

> **只有一个线程能成功把 `tail_` 从 10 改成 11。**

失败的线程会重新读新值，再抢下一次。

所以 CAS 的核心作用是：

> **无锁地抢占一个唯一位置。**

---

# 6. 代码里 CAS 是怎么用的

## 入队时抢写位置

```cpp
do {
  new_tail = old_tail + 1;
  if (GetIndex(new_tail) == GetIndex(head_.load(...))) {
    return false;
  }
} while (!tail_.compare_exchange_weak(old_tail, new_tail, ...));
```

意思是：

* 我想把 `tail_` 从 `old_tail` 推到 `new_tail`
* 如果中途别人改了，我就失败，再来一次
* 直到我成功占到一个写槽位

---

## 出队时抢读位置

```cpp
do {
  new_head = old_head + 1;
  if (new_head == commit_.load(...)) {
    return false;
  }
  *element = pool_[GetIndex(new_head)];
} while (!head_.compare_exchange_weak(old_head, new_head, ...));
```

意思是：

* 我想把 `head_` 从 `old_head` 推到 `new_head`
* 如果别人已经先把这个位置消费了，我就重试

所以：

* 生产者用 CAS 抢写位
* 消费者用 CAS 抢读位

---

# 7. 为什么说这是“无锁”思路

因为这里没有 `mutex` 这种大锁，而是靠：

* 原子变量
* CAS 重试

来协调线程。

优点是：

* 快
* 冲突小时延低
* 不容易因为大锁阻塞一大片线程

缺点是：

* 实现更难理解
* 代码更容易写错
* 高竞争下会不停重试

这类代码在中间件、消息队列、实时系统里很常见。

---

# 8. `WaitStrategy` 是干什么的

它解决的是另一个问题：

> 队列空了或满了以后，线程怎么办？

如果不用等待机制，你只能这样写：

```cpp
while (!Enqueue(x)) {}
```

或者：

```cpp
while (!Dequeue(&x)) {}
```

这叫**忙等**，会疯狂占 CPU。

所以它引入 `WaitStrategy`，在失败时先等等。

---

## 在代码里怎么体现

### 等待入队

```cpp
while (!break_all_wait_) {
  if (Enqueue(element)) {
    return true;
  }
  if (wait_strategy_->EmptyWait()) {
    continue;
  }
  break;
}
```

意思是：

* 先试着入队
* 失败了，等一会
* 醒来再试
* 超时或被打断就返回失败 

### 等待出队

`WaitDequeue()` 也是一样逻辑。

---

## `NotifyOne()` 是干什么的

入队成功后会：

```cpp
wait_strategy_->NotifyOne();
```

意思是：

> “我放进去一个数据了，可能有等待的消费者，可以叫醒一个。” 

所以 `WaitStrategy` 本质上是：

* 失败时怎么等
* 成功后怎么唤醒别人

---

# 9. `BreakAllWait()` 又是干什么的

这个一般用于**停止系统 / 关闭线程**。

```cpp
break_all_wait_ = true;
wait_strategy_->BreakAllWait();
```

作用是：

> 把所有正在等待入队/出队的线程都打断，让它们赶紧退出等待。

否则线程可能一直卡在等待状态，程序没法优雅退出。

---

# 10. 三者是怎么配合的

你可以把它们看成三层：

## 第一层：环形数组 `pool_`

真正存数据。

## 第二层：`head_ / tail_ / commit_`

描述“哪能写、哪能读、哪已经写完”。

## 第三层：CAS + WaitStrategy

* CAS：解决多线程抢位置
* WaitStrategy：解决空/满时等待

所以这个队列不是简单“数组+两个下标”，而是：

> **环形数组 + 原子状态 + 无锁竞争 + 等待唤醒**

---

# 11. 给你一个完整时序例子

假设队列一开始：

* `head_ = 0`
* `tail_ = 1`
* `commit_ = 1`

现在线程 P 入队一个元素：

### 第一步

P 用 CAS 抢到 `tail_`，把它从 1 改成 2。
这表示：槽位 1 归我写。

### 第二步

P 执行：

```cpp
pool_[GetIndex(1)] = element;
```

真正把数据写进去。

### 第三步

P 把 `commit_` 从 1 改成 2。
这表示：1 号槽位现在正式可读。

---

然后线程 C 出队：

### 第一步

它看：

```cpp
new_head = old_head + 1 = 1
```

并检查：

```cpp
new_head == commit_ ?
```

现在 `commit_ = 2`，不相等，说明 1 号槽可读。

### 第二步

它读：

```cpp
*element = pool_[GetIndex(1)];
```

### 第三步

它 CAS 把 `head_` 从 0 改成 1。
表示：1 号槽已经消费掉了。

---

# 12. 你学习这类代码时的正确切入点

你不要一上来就抠内存序细节。先分三步学：

### 第一步

先把它当普通环形队列，理解 `head/tail`

### 第二步

再理解为什么多了个 `commit`
这是为了区分“占位”和“写完”

### 第三步

再理解 CAS 和等待机制
一个负责竞争，一个负责阻塞/唤醒

这样就不会乱。

---

# 13. 你现在最该记住的结论

这三句话先牢牢记住：

1. **`tail_` 不等于可读边界，`commit_` 才是可读边界**
2. **CAS 用来让多个线程安全地抢一个唯一槽位**
3. **`WaitStrategy` 用来避免空转，负责等待和唤醒**

---

# 14. 下一步最适合学什么

顺着这份代码，最适合下一步学的是：

> **为什么 `Enqueue()` 里先改 `tail_`，再写数据，最后改 `commit_`，这个顺序为什么不能乱。**

这个一旦懂了，你对并发队列会再上一个台阶。
